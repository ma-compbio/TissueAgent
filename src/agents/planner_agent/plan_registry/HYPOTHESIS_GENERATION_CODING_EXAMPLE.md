# Coding Agent Implementation Guide for Hypothesis Generation

This guide shows how the Coding Agent should implement the enhanced dataset inventory and cross-validation steps for the PAPER_HYPOTHESIS_GENERATION plan.

## Step 1: Enhanced Data Inventory with Categorical Values

The Coding Agent should now generate `tables/categorical_values.tsv` in addition to the existing `data_inventory.tsv`.

### Example Code for Categorical Values Export

```python
import anndata as ad
import pandas as pd
from pathlib import Path

# Load dataset
adata_path = DATA_DIR / "dataset" / "overall_merfish.h5ad"
adata = ad.read_h5ad(str(adata_path))

# Extract categorical columns with their unique values
categorical_data = []

for col in adata.obs.columns:
    if adata.obs[col].dtype.name == 'category' or adata.obs[col].dtype == 'object':
        unique_vals = sorted(adata.obs[col].unique().astype(str))
        n_unique = len(unique_vals)

        categorical_data.append({
            'column': col,
            'n_unique': n_unique,
            'unique_values': ','.join(unique_vals),
            'dtype': str(adata.obs[col].dtype),
            'semantic_hint': infer_semantic_type(col)  # helper function
        })

# Save categorical values
cat_df = pd.DataFrame(categorical_data)
cat_path = DATA_DIR / "tables" / "categorical_values.tsv"
cat_df.to_csv(cat_path, sep='\t', index=False)

print(f"Saved: {cat_path}")
print(f"Categorical columns documented: {len(categorical_data)}")
```

### Helper Function for Semantic Hints

```python
def infer_semantic_type(col_name):
    """Infer the semantic meaning of a column based on its name."""
    col_lower = col_name.lower()

    if 'cluster' in col_lower and 'zone' in col_lower:
        return "cluster_id_not_anatomical_depth"
    elif 'communit' in col_lower:
        return "anatomical_region_label"
    elif 'population' in col_lower or 'cell_type' in col_lower:
        return "cell_type_annotation"
    elif 'sample' in col_lower or 'batch' in col_lower:
        return "sample_identifier"
    elif 'leiden' in col_lower:
        return "unsupervised_cluster_id"
    else:
        return "unspecified"
```

## Step 2: Paper-Dataset Cross-Validation

The Coding Agent should generate `tables/paper_dataset_reconciliation.tsv` by comparing paper entities with actual dataset values.

### Example Code for Cross-Validation

```python
import json
from pathlib import Path

# Load paper entities (generated by PDF Reader Agent)
entities_path = DATA_DIR / "tables" / "entities_and_markers.json"
with open(entities_path) as f:
    paper_entities = json.load(f)

# Load dataset
adata_path = DATA_DIR / "dataset" / "overall_merfish.h5ad"
adata = ad.read_h5ad(str(adata_path))

# Extract actual values from dataset
dataset_communities = set(adata.obs['communities'].unique())
dataset_populations = set(adata.obs['populations'].unique())
dataset_genes = set(adata.var_names)

reconciliation = []

# Check communities from paper
for comm in paper_entities.get('communities', []):
    paper_name = comm['name']

    # Try exact match (case-insensitive)
    exact_match = None
    for ds_comm in dataset_communities:
        if paper_name.lower().replace('-', ' ') == ds_comm.lower().replace('-', ' '):
            exact_match = ds_comm
            break

    if exact_match:
        reconciliation.append({
            'entity_type': 'community',
            'paper_name': paper_name,
            'dataset_column': 'communities',
            'dataset_value': exact_match,
            'status': 'EXACT_MATCH',
            'alternative': None,
            'note': f"Paper community '{paper_name}' found as '{exact_match}'"
        })
    else:
        # Check if composition suggests alternative
        composition = comm.get('approximate_composition', [])
        alternatives = suggest_alternatives(paper_name, composition, dataset_populations, dataset_communities)

        reconciliation.append({
            'entity_type': 'community',
            'paper_name': paper_name,
            'dataset_column': 'communities',
            'dataset_value': None,
            'status': 'MISSING',
            'alternative': alternatives,
            'note': f"Paper community '{paper_name}' not found in dataset. Suggest: {alternatives}"
        })

# Check cell populations from paper
for cell_category, cell_list in paper_entities.get('cells', {}).items():
    for cell_type in cell_list:
        exact_match = None
        for ds_pop in dataset_populations:
            if cell_type.lower().replace('-', ' ') == ds_pop.lower().replace('-', ' '):
                exact_match = ds_pop
                break

        if exact_match:
            reconciliation.append({
                'entity_type': 'cell_type',
                'paper_name': cell_type,
                'dataset_column': 'populations',
                'dataset_value': exact_match,
                'status': 'EXACT_MATCH',
                'alternative': None,
                'note': f"Cell type '{cell_type}' found"
            })
        else:
            fuzzy_matches = find_fuzzy_matches(cell_type, dataset_populations)
            status = 'FUZZY_MATCH' if fuzzy_matches else 'MISSING'

            reconciliation.append({
                'entity_type': 'cell_type',
                'paper_name': cell_type,
                'dataset_column': 'populations',
                'dataset_value': fuzzy_matches[0] if fuzzy_matches else None,
                'status': status,
                'alternative': ','.join(fuzzy_matches) if fuzzy_matches else None,
                'note': f"Fuzzy match: {fuzzy_matches}" if fuzzy_matches else "Not found"
            })

# Check markers from paper
for marker in paper_entities.get('markers', []):
    if marker in dataset_genes:
        # Get expression stats
        marker_expr = adata[:, marker].X
        nonzero_frac = (marker_expr > 0).sum() / len(marker_expr)

        reconciliation.append({
            'entity_type': 'marker',
            'paper_name': marker,
            'dataset_column': 'var_names',
            'dataset_value': marker,
            'status': 'PRESENT',
            'alternative': None,
            'note': f"Gene present; {nonzero_frac:.1%} cells express"
        })
    else:
        reconciliation.append({
            'entity_type': 'marker',
            'paper_name': marker,
            'dataset_column': 'var_names',
            'dataset_value': None,
            'status': 'ABSENT',
            'alternative': None,
            'note': "Gene not in MERFISH panel"
        })

# Save reconciliation
recon_df = pd.DataFrame(reconciliation)
recon_path = DATA_DIR / "tables" / "paper_dataset_reconciliation.tsv"
recon_df.to_csv(recon_path, sep='\t', index=False)

print(f"Saved: {recon_path}")

# Print summary
print("\nReconciliation Summary:")
print(recon_df['status'].value_counts().to_string())

# Flag critical mismatches
critical = recon_df[(recon_df['entity_type'] == 'community') & (recon_df['status'] == 'MISSING')]
if not critical.empty:
    print("\n⚠️ CRITICAL: Missing communities from paper:")
    for _, row in critical.iterrows():
        print(f"  - {row['paper_name']}: {row['note']}")
```

### Helper Functions

```python
def suggest_alternatives(paper_name, composition, dataset_populations, dataset_communities):
    """Suggest how to reconstruct missing entity."""
    suggestions = []

    # Check if composition cell types exist
    if composition:
        found_types = [c for c in composition if any(c.lower() in p.lower() for p in dataset_populations)]
        if found_types:
            suggestions.append(f"Filter populations for: {', '.join(found_types)}")

    # Check if it's a spatial concept
    if 'intermediate' in paper_name.lower() or 'boundary' in paper_name.lower():
        suggestions.append("Use spatial coords to identify boundary/transition zones")

    # Check if related communities exist
    related = [c for c in dataset_communities if any(word in c.lower() for word in paper_name.lower().split('-'))]
    if related:
        suggestions.append(f"Related communities found: {', '.join(related)}")

    return ' | '.join(suggestions) if suggestions else 'No clear alternative'

def find_fuzzy_matches(paper_name, dataset_values, threshold=0.7):
    """Find fuzzy string matches."""
    from difflib import SequenceMatcher

    matches = []
    for ds_val in dataset_values:
        similarity = SequenceMatcher(None, paper_name.lower(), ds_val.lower()).ratio()
        if similarity >= threshold:
            matches.append((ds_val, similarity))

    # Sort by similarity, return top matches
    matches.sort(key=lambda x: x[1], reverse=True)
    return [m[0] for m in matches[:3]]
```

## Step 3: Update Hypothesis Agent to Use Reconciliation

The Hypothesis Agent prompt should be updated to read and respect the reconciliation table:

```python
# In hypothesis_agent/prompt.py, add after reading data_inventory:

<execute>
# Read paper-dataset reconciliation
recon_path = DATA_DIR / "tables" / "paper_dataset_reconciliation.tsv"
if recon_path.exists():
    recon_df = pd.read_csv(recon_path, sep='\t')

    # Filter to only validated entities
    validated_communities = recon_df[
        (recon_df['entity_type'] == 'community') &
        (recon_df['status'] == 'EXACT_MATCH')
    ]['dataset_value'].tolist()

    validated_markers = recon_df[
        (recon_df['entity_type'] == 'marker') &
        (recon_df['status'] == 'PRESENT')
    ]['dataset_value'].tolist()

    missing_communities = recon_df[
        (recon_df['entity_type'] == 'community') &
        (recon_df['status'] == 'MISSING')
    ]['paper_name'].tolist()

    print(f"✓ Validated communities: {validated_communities}")
    print(f"✓ Validated markers: {len(validated_markers)}")
    print(f"⚠️ Missing communities: {missing_communities}")

    # Flag if critical entities are missing
    if missing_communities:
        print(f"\nWARNING: Cannot test hypotheses requiring: {missing_communities}")
        print("Hypotheses should use only validated communities or suggest reconstruction")
else:
    print("WARNING: No reconciliation table found!")
    validated_communities = []
    validated_markers = []
</execute>
```

## Summary

With these changes, the workflow becomes:

1. **PDF Reader Agent** extracts entities → `tables/entities_and_markers.json`
2. **Coding Agent** inventories dataset → `tables/data_inventory.tsv` + `tables/categorical_values.tsv`
3. **Coding Agent** cross-validates → `tables/paper_dataset_reconciliation.tsv`
4. **Hypothesis Agent** reads all tables and generates hypotheses using **ONLY validated entities**

This prevents generating untestable hypotheses like "Compare Inner-LV, Intermediate-LV, and Outer-LV" when Intermediate-LV doesn't exist in the dataset.
